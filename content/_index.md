#!/bin/bash

# First argument must be a filename.
File=$1

if [ -z $1 ] ; then
        echo "Please enter a filename!"
else
        if [ -s $File ] ; then
                echo "'$File' exists and contains data."
        elif [ -e $File ] ; then
                echo "'$File' exists and is empty."
        else
                echo "'$File' does not exist."
        fi
fi
